SET SERVEROUTPUT ON
DECLARE 
CURSOR par IS
SELECT COURSE_NO, SECTION_NO, SECTION_ID
FROM SECTION;

CURSOR ch IS
SELECT SECTION_ID
FROM ENROLLMENT;

v_num NUMBER := 0;
var NUMBER;
i NUMBER;

BEGIN
FOR var IN par LOOP
  v_num := 0;
  FOR i IN ch LOOP
    IF (i.SECTION_ID = var.SECTION_ID) 
          THEN v_num := v_num+1; 
          END IF;
  END LOOP;
    DBMS_OUTPUT.PUT_LINE( var.COURSE_NO || ' ' || var.SECTION_NO || ' ' || v_num );
  END LOOP;
END;
